temps = [22, 25, 19, 30, 28, 24, 21]
print(sum(temps) / len(temps))

count = 0
for t in temps:
    if t >= 25:
        count += 1
print(count, '日')

prices = {"りんご": 120, "みかん": 80, "バナナ": 100}
