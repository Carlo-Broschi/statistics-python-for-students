import numpy as np, matplotlib.pyplot as plt
from statsmodels.duration.survfunc import SurvfuncRight

# 問1: 月間利用「少(≤5)/多(>5)」でカプランマイヤーを比較
surv['利用'] = np.where(surv['月間利用'] > 5, '多', '少')
fig, ax = plt.subplots(figsize=(7,4.5))
for grp, sub in surv.groupby('利用'):
    s = SurvfuncRight(sub['継続月数'], sub['解約'])
    ax.step(s.surv_times, s.surv_prob, where='post', label=f'利用{grp}')
ax.set_ylim(0,1); ax.legend(); ax.set_xlabel('継続月数'); ax.set_ylabel('継続率'); plt.show()
# → よく使う顧客ほど継続率が高い（解約しにくい）。
