for grp, row in g.iterrows():
    a = 10 + row['申込数']; b = 90 + (row['訪問数'] - row['申込数'])
    post2[grp] = stats.beta(a, b)

    print(grp, '事後平均', round(d.mean(), 4))
# → データが多いと事前(10,90)の影響は小さく、Beta(1,1)とほぼ同じ結論。
